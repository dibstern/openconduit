interface EventRouter { input: ReadableStream<SSEEvent>; channels: { session: Channel<SessionEvent>; message: Channel<MessageEvent>; system: Channel<SystemEvent>; }; } function routeEvent(event: SSEEvent): ChannelTarget { switch (event.type) { case "session.updated": case "session.created": return "session"; case "message.delta": return "message"; default: return "system"; } }

The routeEvent function examines each incoming event and routes it to the appropriate output channel based on its type prefix.
