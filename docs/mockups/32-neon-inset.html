interface EventConduit { input: ReadableStream<SSEEvent>; channels: { session: Channel<SessionEvent>; message: Channel<MessageEvent>; system: Channel<SystemEvent>; }; } function route(event: SSEEvent): ChannelTarget { switch (event.type) { case "session.updated": case "session.created": return "session"; case "message.delta": return "message"; default: return "system"; } }

The route function acts as the conduit — examining each incoming event and routing it to the appropriate output channel.
